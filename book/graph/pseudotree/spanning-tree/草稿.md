// 找环，存在circle数组中
int find_circle(int u, int fa) {
    if(vis[u]) return u;
    vis[u] = 1;
    for(int i = h[u]; ~i ; i = ne[i]) {
        int j = e[i];
        if(j == fa) continue;
        int t = find_circle(j, u);
        if(t) {
            //vis=2 标记环上的点
            circle[++ top] = u;
            v[top] = w[i]; // 记录边上的权值
            vis[u] = 2;
            return t == u ? 0 : t;
        }
    }
    return 0;
}
